### Trend in Restaurant Inspections by Grade
```{r plot1}
astoria_inspec %>%
  drop_na(grade) %>%
  group_by(year, grade) %>%
  summarise(
    inspections = n()
  ) %>%
  ungroup() %>%
  plot_ly(x = ~year, y = ~inspections, color = ~grade, alpha = .5,
          type = "scatter", mode = "lines", colors = "viridis") 
```


### Proportion of Critical Inspections by Select Cuisine Types

```{r plot2}
astoria_inspec %>%
  group_by(cuisine_description, critical_flag) %>%
  summarise(inspections = n()) %>%
  ungroup() %>%
  pivot_wider(
    names_from = critical_flag,
    values_from = inspections
  ) %>%
  janitor::clean_names() %>%
  mutate(
    total = critical + not_critical,
    percent_critical = critical / total
    ) %>%
  filter(total > 200) %>%
  mutate(cuisine_description = fct_reorder(as.factor(cuisine_description), percent_critical)) %>%
  plot_ly(x = ~cuisine_description, y = ~percent_critical, color = ~cuisine_description, alpha = .5,
          type = "bar", colors = "viridis") %>% 
  layout(showlegend = FALSE)
```
